삽질기

  1. keytool 명령어가 keystore 인줄 잘못 봤음
  2. keytool 은 jdk 에 있었음 (anroid sdk 에 있는 줄)
  3. 암호가 6자는 넘어야함
// :\Java\jdk1.8.0_191\bin\keytool -genkey -v -keystore dcapp.keystore -alias dcapp -keyalg RSA -keysize 2048 -validity 10000
PlayerSettings.Android.keystoreName = "Assets/Editor/dcapp.keystore";
PlayerSettings.Android.keystorePass = "dcappkey";
PlayerSettings.Android.keyaliasName = "dcapp";
PlayerSettings.Android.keyaliasPass = "dcappkey";
